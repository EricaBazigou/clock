import tkinter as tk
from time import strftime

def update_time():
    """
    Παίρνει την τρέχουσα ώρα και ενημερώνει την ετικέτα του ρολογιού.
    Στη συνέχεια, καλεί τον εαυτό της μετά από 1000ms (1 δευτερόλεπτο).
    """
    # Μορφοποίηση ώρας: %H=Ώρα (24ωρο), %M=Λεπτά, %S=Δευτερόλεπτα
    current_time = strftime('%H:%M:%S')
    
    # Ενημέρωση του κειμένου της ετικέτας
    clock_label.config(text=current_time)
    
    # Προγραμματίζει την επανάληψη της συνάρτησης μετά από 1000ms
    # Η λειτουργία after() είναι απαραίτητη για την ενημέρωση σε πραγματικό χρόνο
    clock_label.after(1000, update_time)

# --- Δημιουργία Κύριου Παραθύρου ---
root = tk.Tk()
root.title("Το πρώτο μου ρολόι")
# Μπορείς να ορίσεις μέγεθος, αλλά το Tkinter θα προσαρμοστεί αυτόματα.

# --- Δημιουργία και Διαμόρφωση Ετικέτας (Label) ---
clock_label = tk.Label(
    root, 
    font=('calibri', 60, 'bold'), # Γραμματοσειρά και μέγεθος
    background='black',          # Χρώμα φόντου
    foreground='pink'            # Χρώμα κειμένου
)

# Τοποθέτηση της ετικέτας στο κέντρο του παραθύρου
clock_label.pack(anchor='center', fill='both', expand=1, padx=20, pady=20)

# --- Εκκίνηση Ρολογιού και Κύριου Βρόχου ---
update_time() # Αρχική κλήση για να ξεκινήσει το ρολόι
root.mainloop() # Εκκινεί τον κύριο βρόχο του Tkinter (το GUI παραμένει ανοιχτό)
